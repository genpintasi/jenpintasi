function switchContent(level) {
  const data = contentData[level];
  
  document.getElementById('hero-title').innerText = data.heroTitle;
  document.getElementById('feature-title').innerText = data.featureTitle;
  document.getElementById('feature-intro').innerText = data.featureIntro;
  document.getElementById('feature-iframe').src = data.featureIframe;

  document.getElementById('selection-title').innerText = data.selectionTitle;
  document.getElementById('selection-iframe').src = data.selectionIframe;
  
  const exclusiveSec = document.getElementById('exclusive-section');
  if (level === 'primary') {
    exclusiveSec.style.display = 'block';
  } else {
    exclusiveSec.style.display = 'none';
  }

  const selectionGrid = document.getElementById('selection-grid');
  selectionGrid.innerHTML = data.selectionReasons.map(r => `
    <div class="card">
      <img src="${r.img}" alt="${r.title}">
      <h3>${r.title}</h3>
      <p>${r.desc}</p>
    </div>
  `).join('');

  document.getElementById('ebook-note').innerText = data.ebookNote;
  document.getElementById('ebook-iframe').src = data.ebookIframe;
  document.getElementById('video-intro').innerText = data.videoIntro;
  document.getElementById('video-iframe').src = data.videoSrc;

  document.getElementById('resource-intro').innerText = data.resourceIntro;
  const resourceGrid = document.getElementById('resource-grid');
  resourceGrid.innerHTML = data.resources.map(r => `
    <div class="res-card">
      <a href="${r.link}" target="_blank" style="text-decoration:none; color:inherit;">
        <img src="${r.img}" alt="${r.title}">
        <h3><span class="resource-badge">${r.badge}</span><br>${r.title}</h3>
        <p>${r.desc}</p>
      </a>
    </div>
  `).join('');

  document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));
  document.getElementById('btn-' + level).classList.add('active');
  
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

window.onload = () => switchContent('primary');
